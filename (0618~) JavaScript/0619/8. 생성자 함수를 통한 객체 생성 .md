  <script>
    //객체 생성을 위한 생성자 함수 정의
    function Student (name, ko, math, en, sci){
      this.name = name;
      this.ko=ko;
      this.math=math;
      this.en=en;
      this.sci=sci;
    }
      Student.prototype.total= function (){
        return this.ko + this.math + this.en + this.sci;

      },
      Student.prototype.average = function (){
      return this.total()/4;
      }

    console.dir(Student);
