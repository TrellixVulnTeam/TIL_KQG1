tsvRead.js

```js
window.addEventListener("load", function() {
	// 1.데이터준비
	d3.tsv("data.tsv")
	.then(function(data) {
		console.log(data)
		var dataSet = [];
		for (var i = 0; i < data.length; i++) {
			dataSet.push(data[i].item1);
		}
	d3.select("#myGraph")
	.selectAll("rect")
	.data(dataSet) // 데이터 설정
	.enter() // 데이터 수에 따라서 rect요소 생성
	.append("rect")
	.attr("class", "bar")
	.attr("width", function(d, i) { // .data(dataSet)설정으로
		// d,i생김
		return d;
	})
	.attr("height",20)
	.attr("x",0)
	.attr("y",function(d,i){
		return i*25;
	})
})
});
```
